    <script>
        function validateform() {
            

		var name = document.regform.name.value.trim();
		var mail = document.regform.mail.value.trim();
            var password = document.regform.pwd.value.trim();
            var pno = document.regform.pno.value.trim();
            var age = document.regform.age.value.trim();

		  if (name === "") {
                alert("Name cannot be blank");
                return false;
            }
		if (mail === "") {
                alert("Email cannot be blank");
                return false;
            }
            if (password.length < 6 || password === "") {
                alert("Password must be at least 6 characters long");
                return false;
            }
            
            if (pno.length !== 10 || isNaN(pno) || pno === "") {
                alert("Phone number must be exactly 10 digits");
                return false;
            }
            if (isNaN(age) || age < 11 || age > 60 || age === "") {
                alert("Age must be between 11 and 60");
                return false;
            }
		return true;
        }
    </script>
